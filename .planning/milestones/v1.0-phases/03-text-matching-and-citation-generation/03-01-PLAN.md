<objective>
Build the text matching engine and citation formatter for granted patents. Given selected text from Google Patents HTML, normalize it, find it in the PositionMap via exact-then-fuzzy matching, and return a formatted column:line citation with confidence score.

Purpose: This is the core matching pipeline that connects user-selected text to PDF position data. It handles the known HTML-to-PDF text divergences (smart quotes, dashes, ligatures, whitespace) through normalization, then locates the selection in the PositionMap using a concatenation + sliding window approach.

Output: A working citation lookup pipeline where content script sends selected text to service worker, which routes to offscreen document, which retrieves PositionMap from IndexedDB, runs matching, and returns a formatted citation string with confidence score.

**2. matchAndCite(selectedText, positionMap)**

Core matching function. Returns `{ citation, startEntry, endEntry, confidence }` or `null` if no match.

Algorithm:
1. Normalize the selected text
2. Build concatenated PositionMap text with entry boundary tracking:
   - For each entry, normalize its text
   - If previous entry has `hasWrapHyphen`, strip the trailing hyphen from the previous text and concatenate without space
   - Otherwise, insert a single space between entries
   - Track `{ charStart, charEnd, entryIdx }` boundaries for each entry
3. Exact match: `concat.indexOf(normalizedSelection)`
   - If found, map character positions back to PositionMap entries via boundaries
   - Return with confidence 1.0
4. Fuzzy fallback: `fuzzySubstringMatch(normalizedSelection, concat)`
   - Only if exact match fails
   - Allow up to 20% edit distance
   - Return with computed similarity score
5. Map match boundaries to start/end PositionMap entries
6. Format citation using `formatCitation(startEntry, endEntry)`

```javascript
function matchAndCite(selectedText, positionMap) {
  const normalized = normalizeText(selectedText);
  if (!normalized || normalized.length < 2) return null;

  // Build concatenated text with boundary tracking
  let concat = '';
  const boundaries = [];

  for (let i = 0; i < positionMap.length; i++) {
    const entry = positionMap[i];
    let lineText = normalizeText(entry.text);

    // Handle wrap hyphens: strip trailing hyphen, no space before next
    const prevEntry = i > 0 ? positionMap[i - 1] : null;
    if (prevEntry && prevEntry.hasWrapHyphen) {
      // Previous line was hyphen-wrapped; this line continues the word
      // Don't add space (previous hyphen was already stripped)
    } else if (concat.length > 0) {
      concat += ' ';
    }

    const charStart = concat.length;
    concat += lineText;

    boundaries.push({
      charStart,
      charEnd: concat.length,
      entryIdx: i,
    });
  }

  // Strip wrap hyphens from concatenated text
  // We need to rebuild to handle this correctly:
  // Actually, rebuild the concat properly:
  concat = '';
  for (let i = 0; i < positionMap.length; i++) {
    const entry = positionMap[i];
    let lineText = normalizeText(entry.text);

    if (entry.hasWrapHyphen) {
      // Remove trailing hyphen (it's a wrap hyphen, not real)
      lineText = lineText.replace(/-$/, '');
    }

    if (i > 0 && !positionMap[i - 1].hasWrapHyphen) {
      concat += ' ';
    }

    boundaries[i].charStart = concat.length;
    concat += lineText;
    boundaries[i].charEnd = concat.length;
  }

  // Exact normalized match
  const idx = concat.indexOf(normalized);
  if (idx !== -1) {
    return resolveMatch(idx, idx + normalized.length, boundaries, positionMap, 1.0);
  }

  // Fuzzy fallback
  const fuzzyResult = fuzzySubstringMatch(normalized, concat);
  if (fuzzyResult && fuzzyResult.similarity >= 0.80) {
    return resolveMatch(
      fuzzyResult.start, fuzzyResult.end,
      boundaries, positionMap, fuzzyResult.similarity
    );
  }

  return null; // No match
}
```

**3. resolveMatch(matchStart, matchEnd, boundaries, positionMap, confidence)**

Map character-level match positions back to PositionMap entries:

```javascript
function resolveMatch(matchStart, matchEnd, boundaries, positionMap, confidence) {
  const startBoundary = boundaries.find(b => b.charStart <= matchStart && b.charEnd > matchStart);
  const endBoundary = boundaries.find(b => b.charStart < matchEnd && b.charEnd >= matchEnd);

  if (!startBoundary || !endBoundary) return null;

  const startEntry = positionMap[startBoundary.entryIdx];
  const endEntry = positionMap[endBoundary.entryIdx];

  return {
    citation: formatCitation(startEntry, endEntry),
    startEntry,
    endEntry,
    confidence,
  };
}
```

**4. formatCitation(startEntry, endEntry)**

Format citation per locked decisions:

```javascript
function formatCitation(startEntry, endEntry) {
  const startCol = startEntry.column;
  const startLine = startEntry.lineNumber;
  const endCol = endEntry.column;
  const endLine = endEntry.lineNumber;

  if (startCol === endCol && startLine === endLine) {
    return `${startCol}:${startLine}`;
  } else if (startCol === endCol) {
    return `${startCol}:${startLine}-${endLine}`;
  } else {
    return `${startCol}:${startLine}-${endCol}:${endLine}`;
  }
}
```

- Single line: `4:15`
- Same-column range: `4:15-20`
- Cross-column range: `4:55-5:10`
- Claims and description use identical col:line format (no prefix distinction)

**5. fuzzySubstringMatch(needle, haystack)**

Levenshtein-based sliding window match as fallback:

```javascript
function fuzzySubstringMatch(needle, haystack) {
  const n = needle.length;
  if (n === 0) return null;
  const maxDistance = Math.floor(n * 0.2); // Allow 20% edits

  let bestSimilarity = 0;
  let bestStart = -1;
  let bestEnd = -1;

  const windowMin = Math.max(1, n - maxDistance);
  const windowMax = n + maxDistance;

  for (let windowSize = windowMin; windowSize <= windowMax; windowSize++) {
    for (let start = 0; start <= haystack.length - windowSize; start++) {
      const candidate = haystack.substring(start, start + windowSize);
      const distance = levenshtein(needle, candidate);
      const similarity = 1 - distance / Math.max(n, windowSize);

      if (similarity > bestSimilarity) {
        bestSimilarity = similarity;
        bestStart = start;
        bestEnd = start + windowSize;
      }
    }
  }

  if (bestSimilarity >= 0.80) {
    return { start: bestStart, end: bestEnd, similarity: bestSimilarity };
  }
  return null;
}
```

**Performance note:** The fuzzy sliding window is O(n*m*w). For typical selections (50-500 chars) against typical patents (5000-20000 chars concatenated), this is acceptable. For very large selections (>500 chars), the exact normalized match should succeed in most cases, avoiding the fuzzy path entirely.

Include the standard Levenshtein distance function:

```javascript
function levenshtein(a, b) {
  const m = a.length;
  const n = b.length;
  const dp = Array.from({ length: m + 1 }, () => new Array(n + 1).fill(0));
  for (let i = 0; i <= m; i++) dp[i][0] = i;
  for (let j = 0; j <= n; j++) dp[0][j] = j;
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      const cost = a[i - 1] === b[j - 1] ? 0 : 1;
      dp[i][j] = Math.min(dp[i - 1][j] + 1, dp[i][j - 1] + 1, dp[i - 1][j - 1] + cost);
    }
  }
  return dp[m][n];
}
```

```javascript
/**
 * Look up selected text in the stored PositionMap.
 *
 * Retrieves the PositionMap from IndexedDB, runs the matching
 * algorithm, and sends the result back via CITATION_RESULT message.
 *
 * Note: The matching functions (normalizeText, matchAndCite, etc.)
 * are duplicated here from text-matcher.js because offscreen documents
 * cannot share classic script globals with content scripts.
 * The offscreen document needs its own copy of these functions.
 */
async function lookupPosition(selectedText, patentId, tabId) {
  try {
    const db = await openDb();
    const record = await new Promise((resolve, reject) => {
      const tx = db.transaction('pdfs', 'readonly');
      const store = tx.objectStore('pdfs');
      const request = store.get(patentId);
      request.onsuccess = () => resolve(request.result);
      request.onerror = (event) => reject(new Error(`IndexedDB read: ${event.target.error}`));
      tx.oncomplete = () => db.close();
    });

    if (!record || !record.positionMap || record.positionMap.length === 0) {
      chrome.runtime.sendMessage({
        type: CITATION_RESULT,
        tabId,
        success: false,
        error: 'no-position-map',
      });
      return;
    }

    const result = matchAndCiteOffscreen(selectedText, record.positionMap);

    if (result) {
      chrome.runtime.sendMessage({
        type: CITATION_RESULT,
        tabId,
        success: true,
        citation: result.citation,
        confidence: result.confidence,
        startEntry: {
          column: result.startEntry.column,
          lineNumber: result.startEntry.lineNumber,
          page: result.startEntry.page,
          section: result.startEntry.section,
        },
        endEntry: {
          column: result.endEntry.column,
          lineNumber: result.endEntry.lineNumber,
          page: result.endEntry.page,
          section: result.endEntry.section,
        },
      });
    } else {
      chrome.runtime.sendMessage({
        type: CITATION_RESULT,
        tabId,
        success: false,
        error: 'no-match',
      });
    }
  } catch (error) {
    console.error('[Offscreen] Lookup error:', error);
    chrome.runtime.sendMessage({
      type: CITATION_RESULT,
      tabId,
      success: false,
      error: error.message,
    });
  }
}
```
