    ```js
    // tests/e2e/lib/pdf-fetch.js
    import fs from 'node:fs';
    import path from 'node:path';
    import { fileURLToPath } from 'node:url';

    const __dirname = path.dirname(fileURLToPath(import.meta.url));
    const CACHE_DIR = path.resolve(__dirname, '../.pdf-cache');

    /**
     * Find the patentimages PDF URL by fetching the Google Patents page and
     * regex-extracting the first patentimages.storage.googleapis.com/...pdf href.
     * Mirrors src/content/content-script.js#findPdfLink conceptually (fresh impl).
     * @param {string} patentId
     * @returns {Promise<string>}
     */
    async function findPdfLink(patentId) {
      const pageUrl = `https://patents.google.com/patent/${patentId}`;
      const res = await fetch(pageUrl);
      if (!res.ok) throw new Error(`Patent page fetch failed: ${res.status} ${pageUrl}`);
      const html = await res.text();
      const m = html.match(/https:\/\/patentimages\.storage\.googleapis\.com\/[^"'\s)]+\.pdf/);
      if (!m) throw new Error(`PDF_LINK_NOT_FOUND: no patentimages PDF href in ${pageUrl}`);
      return m[0];
    }

    /**
     * @param {string} patentId
     * @returns {Promise<string>} absolute path to cached PDF
     */
    export async function ensureCachedPdf(patentId) {
      fs.mkdirSync(CACHE_DIR, { recursive: true });
      const cached = path.join(CACHE_DIR, `${patentId}.pdf`);
      if (fs.existsSync(cached) && fs.statSync(cached).size > 5_000) {
        return cached;
      }
      const pdfUrl = await findPdfLink(patentId);
      const res = await fetch(pdfUrl);
      if (!res.ok) throw new Error(`PDF fetch failed: ${res.status} ${pdfUrl}`);
      const ct = res.headers.get('content-type') || '';
      if (!ct.includes('pdf')) {
        // RESEARCH.md Pitfall 5 — got HTML instead of PDF
        throw new Error(`PDF_FETCH_NOT_PDF: content-type was '${ct}' for ${pdfUrl}`);
      }
      const buf = Buffer.from(await res.arrayBuffer());
      fs.writeFileSync(cached, buf);
      return cached;
    }
    ```

    4. **Function `function inferColumnLine(pageItems, pageWidth, pageHeight)`** — FRESH IMPLEMENTATION mirroring `position-map-builder.js` algorithm:
       - **Filter headers/footers:** drop items where `y > pageHeight - HEADER_TOP_PT` or `y < FOOTER_BOTTOM_PT`.
       - **Detect two-column layout:** count items with `x < pageWidth/2` (leftCount) vs `x >= pageWidth/2` (rightCount). If `min(left,right) / max(left,right) > 0.3`, treat as two-column.
       - **Find column boundary:** in two-column case, build histogram of x-coords across `[pageWidth*0.3, pageWidth*0.7]` middle band in 5pt buckets; the empty/lowest-count bucket is the boundary. In single-column case, boundary = pageWidth.
       - **Cluster into lines:** sort items by `-y` (PDF origin is bottom-left, so descending y = top→bottom). Group consecutive items with `|y_i - y_prev| < Y_LINE_CLUSTER_TOLERANCE_PT` into a single line.
       - **Extract printed column numbers:** scan top-90pt band for items matching `/^\d{1,3}$/` on each side of the column boundary. If both sides found AND `right === left + 1` AND `left % 2 === 1`, use as the printed column pair for that page. Else null (fall back to page-order inference).
       - **Filter gutter line numbers:** drop items matching `/^\d+$/` near the column boundary if the number is a multiple of 5 in range [5, 65] — these are the printed line numbers, not content.
       - **Assign line numbers:** if ≥2 gutter markers were captured on this column, use them as a grid (interpolate y → lineNumber). Else: median-spacing — sort line cluster centers, compute median gap, assign sequential 1..N.
       - Return `{columns: [...], lines: [{col, lineNumber, text, y, items}]}`.
