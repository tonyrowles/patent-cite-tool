    ```js
    // tests/e2e/lib/report.js
    import fs from 'node:fs';
    import path from 'node:path';
    import { fileURLToPath } from 'node:url';
    import { ERROR_CLASSES } from './error-codes.js';

    const __dirname = path.dirname(fileURLToPath(import.meta.url));
    const ARTIFACTS_ROOT = path.resolve(__dirname, '../artifacts');

    /**
     * @param {string} runId
     * @returns {string} absolute path to report.json for this run
     */
    export function reportPathFor(runId) {
      return path.join(ARTIFACTS_ROOT, runId, 'report.json');
    }

    function emptySummary() {
      return {
        total: 0, passed: 0, skipped: 0, failed: 0,
        by_error_class: Object.fromEntries(ERROR_CLASSES.map(k => [k, 0])),
      };
    }

    function emptyReport(runId) {
      return {
        run_id: runId,
        started: new Date().toISOString(),
        ended: null,
        summary: emptySummary(),
        cases: [],
      };
    }

    function readOrInit(reportPath) {
      try {
        return JSON.parse(fs.readFileSync(reportPath, 'utf8'));
      } catch {
        const runId = path.basename(path.dirname(reportPath));
        return emptyReport(runId);
      }
    }

    function recomputeSummary(cases) {
      const s = emptySummary();
      for (const c of cases) {
        s.total++;
        if (c.status === 'passed') s.passed++;
        else if (c.status === 'skipped') s.skipped++;
        else if (c.status === 'failed') s.failed++;
        if (c.errorClass && ERROR_CLASSES.includes(c.errorClass)) {
          s.by_error_class[c.errorClass]++;
        }
      }
      return s;
    }

    /**
     * @param {string} reportPath
     * @param {CaseEntry} caseEntry
     */
    export function appendCase(reportPath, caseEntry) {
      const report = readOrInit(reportPath);
      const idx = report.cases.findIndex(c => c.id === caseEntry.id);
      if (idx >= 0) report.cases[idx] = caseEntry;
      else report.cases.push(caseEntry);
      report.summary = recomputeSummary(report.cases);
      report.ended = new Date().toISOString();
      fs.mkdirSync(path.dirname(reportPath), { recursive: true });
      fs.writeFileSync(reportPath, JSON.stringify(report, null, 2));
    }

    /**
     * @param {string} reportPath
     * @param {object} report
     */
    export function writeReport(reportPath, report) {
      fs.mkdirSync(path.dirname(reportPath), { recursive: true });
      fs.writeFileSync(reportPath, JSON.stringify(report, null, 2));
    }

    export { ERROR_CLASSES };
    ```

    Notes:
    - `recomputeSummary` is correct-by-construction: passes through every case once. Cheaper than maintaining incremental deltas, also handles status-flip on replace (Test 4).
    - `caseEntry.errorClass` of `null`, undefined, or an unknown string is treated as "no error class" for by_error_class purposes but `failed` status still bumps `summary.failed` (Tests 5 + 6). The explicit `ERROR_CLASSES.includes(...)` check enforces this.
    - File-locking is NOT needed — Playwright runs `workers: 1` per Phase 26 (RESEARCH.md §Architecture Patterns Pattern 6). Single-process atomicity is sufficient.
    - The file path resolution mirrors `tests/e2e/lib/artifacts.js` line 30 (`path.resolve(__dirname, '../artifacts')`).
