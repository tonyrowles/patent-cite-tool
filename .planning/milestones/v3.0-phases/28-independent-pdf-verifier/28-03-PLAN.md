    ```js
    // tests/e2e/lib/pdf-snippet.js
    //
    // DIAG-03 — render the PDF page region containing the cited column:line
    // and crop to a tight band around the cited line. Triggered ONLY on
    // VERIFIER_DISAGREE by regression.spec.js. Output is a PNG suitable for
    // a human to scan and decide whether the verifier or the extension is
    // wrong about where the selected text lives.
    //
    // Independence: does NOT import from src/. Pdfjs is consumed via the
    // legacy/build/pdf.mjs entry (RESEARCH.md Pitfall 1). The canvas backend
    // is provided by @napi-rs/canvas via pdfjs's built-in canvasFactory
    // (research-locked; no direct canvas import).

    import fs from 'node:fs';
    import path from 'node:path';
    import { fileURLToPath } from 'node:url';
    import sharp from 'sharp';
    import { getDocument } from 'pdfjs-dist/legacy/build/pdf.mjs';
    import { ensureCachedPdf } from './pdf-fetch.js';

    const __dirname = path.dirname(fileURLToPath(import.meta.url));
    const ARTIFACTS_ROOT = path.resolve(__dirname, '../artifacts');
    const CMAP_URL = path.resolve('node_modules/pdfjs-dist/cmaps/') + '/';
    const STANDARD_FONT_DATA_URL = path.resolve('node_modules/pdfjs-dist/standard_fonts/') + '/';

    // Cache rendered full-page PNG buffers within a single Node process,
    // keyed by `${patentId}:${pageNum}:${dpi}`. Repeated VERIFIER_DISAGREE
    // verdicts on the same patent+page reuse the rendered buffer (cheap;
    // sharp crops are fast).
    const renderCache = new Map();

    function ensureRunDir(runId) {
      const dir = path.join(ARTIFACTS_ROOT, runId);
      fs.mkdirSync(dir, { recursive: true });
      return dir;
    }

    async function renderFullPagePng(patentId, pageNum, dpi) {
      const key = `${patentId}:${pageNum}:${dpi}`;
      if (renderCache.has(key)) return renderCache.get(key);

      const pdfPath = await ensureCachedPdf(patentId);
      const data = new Uint8Array(fs.readFileSync(pdfPath));
      const pdf = await getDocument({
        data,
        cMapUrl: CMAP_URL,
        cMapPacked: true,
        standardFontDataUrl: STANDARD_FONT_DATA_URL,
      }).promise;
      try {
        const page = await pdf.getPage(pageNum);
        const viewport = page.getViewport({ scale: dpi / 72 });
        const canvasFactory = pdf.canvasFactory;
        const cc = canvasFactory.create(viewport.width, viewport.height);
        await page.render({ canvasContext: cc.context, viewport }).promise;
        const pngBuffer = cc.canvas.toBuffer('image/png');
        renderCache.set(key, { pngBuffer, viewportWidth: viewport.width, viewportHeight: viewport.height });
        return renderCache.get(key);
      } finally {
        await pdf.destroy();
      }
    }

    /**
     * Estimate the pixel y-coordinate of the cited line in the rendered page.
     *
     * Strategy: re-parse text content for this page (cheap — pdfjs is already
     * loaded), find the cluster of items at the cited line, and average their
     * pdf-space y. Convert pdf-space y to image-space pixel y:
     *   pixelY = pageHeight_px - (y_pdf * dpi / 72)
     * (PDF origin is bottom-left; PNG origin is top-left, so flip.)
     *
     * Fallback if line not located: estimate from line number assuming
     * 50 lines per page at uniform spacing — produces a snippet near the
     * right region even when column-grid is undetected.
     *
     * @returns {number} pixel y-coordinate (top-left origin) of the cited line center
     */
    async function estimateLinePixelY(patentId, pageNum, citedLine, viewportHeight, dpi) {
      // Reuse the verifier's parser via a fresh pdfjs call (NOT via importing
      // pdf-verifier.js — keep this module orthogonal). Lightweight: getTextContent
      // on a single page is ~50-200ms.
      const pdfPath = await ensureCachedPdf(patentId);
      const data = new Uint8Array(fs.readFileSync(pdfPath));
      const pdf = await getDocument({
        data, cMapUrl: CMAP_URL, cMapPacked: true, standardFontDataUrl: STANDARD_FONT_DATA_URL,
      }).promise;
      try {
        const page = await pdf.getPage(pageNum);
        const textContent = await page.getTextContent();
        const viewport = page.getViewport({ scale: 1.0 });
        // Inline simple line-cluster on this page only (independent of pdf-verifier's
        // full column/line inference — this is "find the y-band that LOOKS like line N")
        const lines = clusterByY(textContent.items, /* tolerancePt */ 3);
        // Sort top→bottom (PDF y descends).
        lines.sort((a, b) => b.y - a.y);
        // citedLine is 1-based; clamp to available lines.
        const idx = Math.min(citedLine - 1, lines.length - 1);
        const target = lines[idx] || lines[lines.length - 1] || { y: viewport.height / 2 };
        // Flip pdf-space y → pixel y at dpi.
        const pixelY = viewportHeight - (target.y * dpi / 72);
        return pixelY;
      } finally {
        await pdf.destroy();
      }
    }

    function clusterByY(items, tolerancePt) {
      const filtered = items.filter(it => it.str && it.str.trim().length > 0);
      filtered.sort((a, b) => b.transform[5] - a.transform[5]);  // top first
      const out = [];
      let cur = null;
      for (const it of filtered) {
        const y = it.transform[5];
        if (!cur || Math.abs(cur.y - y) > tolerancePt) {
          cur = { y, items: [it] };
          out.push(cur);
        } else {
          cur.items.push(it);
        }
      }
      return out;
    }

    /**
     * @param {{patentId: string, page: number, line: number, runId: string, caseId: string}} args
     * @param {{dpi?: number, marginPxAboveLine?: number, marginPxBelowLine?: number}} [opts]
     * @returns {Promise<string>} absolute path to written PNG
     */
    export async function renderPdfSnippet(args, opts = {}) {
      const { patentId, page, line, runId, caseId } = args;
      const dpi = opts.dpi ?? 150;
      const above = opts.marginPxAboveLine ?? 100;
      const below = opts.marginPxBelowLine ?? 100;

      const { pngBuffer, viewportWidth, viewportHeight } = await renderFullPagePng(patentId, page, dpi);
      const lineY = await estimateLinePixelY(patentId, page, line, viewportHeight, dpi);

      // Crop band centered on lineY. Clamp to page bounds.
      const cropTop = Math.max(0, Math.floor(lineY - above));
      const cropBottom = Math.min(Math.floor(viewportHeight), Math.ceil(lineY + below));
      const cropHeight = cropBottom - cropTop;
      const cropWidth = Math.floor(viewportWidth);

      const croppedPng = await sharp(pngBuffer)
        .extract({ left: 0, top: cropTop, width: cropWidth, height: cropHeight })
        .png()
        .toBuffer();

      const outDir = ensureRunDir(runId);
      const outPath = path.join(outDir, `${caseId}-pdf-snippet.png`);
      fs.writeFileSync(outPath, croppedPng);
      return outPath;
    }
    ```

    Constraints:
    - **Must use legacy build** (`pdfjs-dist/legacy/build/pdf.mjs`) per RESEARCH.md Pitfall 1.
    - **Must NOT directly import `@napi-rs/canvas`** — pdfjs's `canvasFactory` lazy-loads it (RESEARCH.md Standard Stack §"`@napi-rs/canvas` ... auto-detected by pdfjs"). If you `import canvas from '@napi-rs/canvas'` you couple to a private API and break Pitfall 6's install-probe assumption.
    - **Must reuse `ensureCachedPdf`** — no duplicate fetch logic. The PDF is already cached after Plan 28-01.
    - **Independence (VFY-02):** Plan 28-04's ESLint rule is scoped to `pdf-verifier.js` only; this file is NOT subject to the rule. Still, by convention this module also does not import src/ — it's part of the verifier-adjacent code path. Keep the bar high.
    - **DPI 150** is the default per CONTEXT.md "Claude's Discretion" recommendation.
    - **Crop margin** is `±100px` per CONTEXT.md.
    - **Cache rendered pages** in `renderCache` (Map) — keyed by `patentId:page:dpi`. RESEARCH.md §"Anti-Patterns to Avoid": "Re-rendering the same page multiple times for snippet — cache the page-render result by patent ID + page number across the run."

    Edge cases:
    - If `line` exceeds the page's actual line count, `estimateLinePixelY` clamps to the last detected line. The snippet may show the footer; that's acceptable — it tells the reviewer "line number was out of range."
    - If `clusterByY` returns 0 lines (blank page or unusual rendering), fall back to `pixelY = viewportHeight / 2` (center crop) — never throw.
    - If `cropHeight <= 0` (line near page edge + small margin), expand the opposite-side margin until `cropHeight > 0`.

    ```js
    // This test FETCHES from Google Patents on first run (then uses cache).
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Node ↔ pdfjs render pipeline | PDF bytes from Google Patents (untrusted) → pdfjs → canvas raster. Pdfjs's job is to be a robust untrusted-input parser; we inherit that guarantee. |
| Node ↔ @napi-rs/canvas native binary | Prebuilt N-API module loaded at runtime. Supply-chain risk surface. |
| Sharp ↔ PNG output | sharp processes the pdfjs-produced PNG buffer. Single-process, trusted input from this point on. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-28-01 | Tampering (supply chain) | @napi-rs/canvas prebuilt binary | mitigate | Already addressed in Plan 28-01 task 28-01-01: exact-pinned in package.json, package-lock.json committed, install-probe runs on every install. This task inherits that mitigation; the snippet renderer just uses what 28-01 verified. |
| T-28-08 | DoS (oversize PDF) | renderPdfSnippet on a malicious huge PDF | accept | The verifier only fetches PDFs from patents.google.com / patentimages.storage.googleapis.com (Plan 28-01 pdf-fetch.js — no arbitrary URL ingestion). Patent PDFs are ≤10MB typical; pdfjs handles this size routinely. CI runner has 7GB RAM; not a meaningful DoS surface. |
| T-28-09 | Information disclosure (snippet PNG contents) | tests/e2e/artifacts/{runId}/{caseId}-pdf-snippet.png | accept | Snippets are slices of public-domain US patent PDFs. No PII, no secrets. `tests/e2e/artifacts/` is already gitignored (Phase 26). |
| T-28-10 | Integrity (silent crop bug → wrong region rendered) | estimateLinePixelY clamping when line out of range | mitigate | Clamping is explicit and documented; the resulting snippet may show footer instead of cited content — this is preferable to throwing (the human reviewer sees "this is the wrong region" and looks at the verifier's `reason` field).
