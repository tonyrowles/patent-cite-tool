       ```js
       // scripts/verify-calibrate.mjs
       // VFY-03 calibration gate. Runs the independent verifier against the
       // 65 currently-passing live cases and asserts ≥95% Tier A/B/C pass rate.
       // RESEARCH.md Pitfall 2 — calibration MUST happen before the verifier is
       // trusted to adjudicate the 10 deferred TIMEOUT_PILL cases.
       import { TEST_CASES } from '../tests/test-cases.js';
       import baseline from '../tests/golden/baseline.json' with { type: 'json' };
       import { verifyCitation } from '../tests/e2e/lib/pdf-verifier.js';

       const TIMEOUT_PILL_DEFERRED_IDS = new Set([
         // Duplicated from regression.spec.js — single source of truth lives in
         // the spec; this script duplicates to avoid importing a Playwright spec.
         // If the spec's list changes, update here too. (Documented in SUMMARY.)
         'US11427642-claims-1', 'US11427642-repetitive', 'US4723129-claims',
         'US5371234-chemical-cross-col', 'US5371234-claims',
         'US7346586-claims-repetitive', 'US8352400-claims',
         'US5440748-claims', 'US5440748-repetitive', 'US4723129-claims-repetitive',
       ]);
       const SYNTHETIC_CATEGORIES = new Set(['gutter']);
       const THROTTLE_MS = 2_000;
       const GATE_PERCENT = 95;

       function patentIdFromCaseId(caseId) {
         const m = caseId.match(/^([A-Z]{2}\d+[A-Z]?\d*)-/);
         if (!m) throw new Error(`Cannot derive patentId from ${caseId}`);
         return m[1];
       }

       async function main() {
         const live = TEST_CASES.filter(tc =>
           !SYNTHETIC_CATEGORIES.has(tc.category) &&
           !TIMEOUT_PILL_DEFERRED_IDS.has(tc.id) &&
           baseline[tc.id]
         );
         console.error(`Calibrating verifier against ${live.length} live cases...`);
         const tierCounts = { A: 0, B: 0, C: 0, D: 0 };
         const failures = [];

         for (const tc of live) {
           const expected = baseline[tc.id].citation;
           const patentId = patentIdFromCaseId(tc.id);
           try {
             const verdict = await verifyCitation({
               patentId, selectedText: tc.selectedText, observedCitation: expected,
             });
             tierCounts[verdict.tier_used]++;
             if (verdict.tier_used === 'D') {
               failures.push({ id: tc.id, citation: expected, reason: verdict.reason });
             }
             console.error(`  [${verdict.tier_used}] ${tc.id} → ${verdict.status}`);
           } catch (e) {
             tierCounts.D++;
             failures.push({ id: tc.id, citation: expected, reason: `EXCEPTION: ${e.message}` });
             console.error(`  [E] ${tc.id} → ${e.message}`);
           }
           await new Promise(r => setTimeout(r, THROTTLE_MS));
         }

         const total = live.length;
         const abc = tierCounts.A + tierCounts.B + tierCounts.C;
         const pct = (abc / total) * 100;
         console.log('\n=== Verifier Calibration Report ===');
         console.log(`Total cases:      ${total}`);
         console.log(`Tier A (exact):   ${tierCounts.A} (${(tierCounts.A/total*100).toFixed(1)}%)`);
         console.log(`Tier B (wsnorm):  ${tierCounts.B} (${(tierCounts.B/total*100).toFixed(1)}%)`);
         console.log(`Tier C (±2 line): ${tierCounts.C} (${(tierCounts.C/total*100).toFixed(1)}%)`);
         console.log(`Tier D (FAIL):    ${tierCounts.D} (${(tierCounts.D/total*100).toFixed(1)}%)`);
         console.log(`Pass rate (A+B+C): ${abc}/${total} = ${pct.toFixed(1)}%`);
         console.log(`Gate: ≥${GATE_PERCENT}%`);
         if (failures.length) {
           console.log('\nTier D failures:');
           for (const f of failures) console.log(`  - ${f.id} (${f.citation}): ${f.reason}`);
         }
         if (pct < GATE_PERCENT) {
           console.error(`\nFAIL: pass rate ${pct.toFixed(1)}% &lt; ${GATE_PERCENT}% gate`);
           process.exit(1);
         }
         console.log('\nPASS: calibration gate met.');
       }

       main().catch(e => { console.error(e); process.exit(2); });
       ```
