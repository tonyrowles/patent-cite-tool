function normalize(s) {
  return s
    .replace(/\s+/g, ' ')           // collapse whitespace
    .replace(/\s*-\s*/g, '-')        // strip space around hyphens
    .replace(/\s+([,;:])/g, '$1')    // strip space before punctuation
    .trim();
}
function normalizeDeep(s) {
  return normalize(s).replace(
    /(?<=[a-zA-Z])([a-zA-Z]) ([a-zA-Z]{4,}-[a-z])/g, '$1$2',  // line-wrap rejoin
  );
}
