    ```javascript
    import { spawn } from 'node:child_process';

    export const LLM_TIMEOUT_MS = 60_000;
    export const SELECTION_MIN_CHARS = 50;
    export const SELECTION_MAX_CHARS = 300;

    const REQUIRED_SELECTION_FIELDS = ['caseId', 'patentId', 'selectedText', 'category', 'rationale'];
    const PATENT_ID_RE = /^[A-Z]{2}\d+[A-Z]?\d*$/;

    /**
     * Spawn `claude -p` with subscription-mode env.
     * Explicitly clears ANTHROPIC_API_KEY (Pitfall 1) and does NOT pass --bare
     * or --json-schema (Pitfalls 1, 2).
     */
    export async function invokeClaudeP({ systemPrompt, userPrompt, timeoutMs = LLM_TIMEOUT_MS } = {}) {
      return new Promise((resolve) => {
        const args = [
          '-p',
          '--output-format', 'json',
          '--max-turns', '1',
          '--system-prompt', systemPrompt,
          userPrompt,
        ];
        const env = { ...process.env, ANTHROPIC_API_KEY: '' };
        const child = spawn('claude', args, { stdio: ['ignore', 'pipe', 'pipe'], env });
        let stdout = '';
        let stderr = '';
        child.stdout.on('data', d => stdout += d.toString());
        child.stderr.on('data', d => stderr += d.toString());

        const timer = setTimeout(() => {
          try { child.kill('SIGTERM'); } catch {}
          resolve({ timedOut: true, stdout: '', stderr, code: null });
        }, timeoutMs);

        child.on('close', (code) => {
          clearTimeout(timer);
          resolve({ timedOut: false, stdout, stderr, code });
        });
        child.on('error', (err) => {
          clearTimeout(timer);
          resolve({ timedOut: false, stdout: '', stderr: err.message, code: -1 });
        });
      });
    }

    /**
     * Parse the raw stdout from claude -p. See RESEARCH.md Pattern 2 for the
     * verified response shape.
     */
    export function parseClaudeResponse(result) {
      const { timedOut, stdout } = result;
      if (timedOut) {
        return { ok: false, errorReason: 'timeout', costUsd: 0, rawSnippet: (result.stderr ?? '').slice(0, 500) };
      }
      if (!stdout || !stdout.trim()) {
        return { ok: false, errorReason: 'empty_stdout', costUsd: 0, rawSnippet: '' };
      }
      let parsed;
      try {
        parsed = JSON.parse(stdout);
      } catch {
        return { ok: false, errorReason: 'json_parse_error', costUsd: 0, rawSnippet: stdout.slice(0, 500) };
      }
      const costUsd = typeof parsed.total_cost_usd === 'number' ? parsed.total_cost_usd : 0;
      if (parsed.is_error) {
        return { ok: false, errorReason: `api_error:${parsed.subtype ?? 'unknown'}`, costUsd, rawSnippet: stdout.slice(0, 500) };
      }
      const modelId = Object.keys(parsed.modelUsage ?? {})[0] ?? 'unknown';
      return {
        ok: true,
        llmText: parsed.result ?? '',
        costUsd,
        modelId,
        durationMs: parsed.duration_ms ?? 0,
        rawJson: parsed,
      };
    }

    /** Build the picker prompt. */
    export function buildPickerPrompt({ patent, specExcerpt, bodyStartPage }) {
      const systemPrompt =
        'You are testing a patent citation extension. Given a patent ID and its specification excerpt, ' +
        'propose ONE textual selection from the specification likely to surface an interesting parser ' +
        'behavior (cross-column boundary, wrap-hyphen, claim-language adjacency, repetitive language). ' +
        'Return STRICT JSON ONLY (no markdown fences), with fields: caseId (string), patentId (string), ' +
        'selectedText (string, verbatim substring of the excerpt, ' + SELECTION_MIN_CHARS + '-' + SELECTION_MAX_CHARS + ' characters), ' +
        'category (one of: modern-short, modern-long, claims, cross-column, repetitive), rationale (one sentence).';
      const userPrompt =
        `Patent: ${patent.id} (category hint: ${patent.category ?? 'unknown'})\n` +
        `Spec excerpt (body starts at page ${bodyStartPage}):\n---\n${specExcerpt}\n---\n` +
        `Return ONLY the JSON object. selectedText must be a verbatim substring of the excerpt above.`;
      return { systemPrompt, userPrompt };
    }

    /** Validate the LLM's JSON output. */
    export function validateLlmSelection(llmText) {
      let obj;
      try {
        obj = JSON.parse(String(llmText ?? '').trim());
      } catch (e) {
        return { ok: false, reason: `JSON parse error: ${e.message}` };
      }
      for (const f of REQUIRED_SELECTION_FIELDS) {
        if (typeof obj[f] !== 'string' || obj[f].length === 0) {
          return { ok: false, reason: `missing or non-string field: ${f}` };
        }
      }
      if (!PATENT_ID_RE.test(obj.patentId)) {
        return { ok: false, reason: `invalid patentId format: ${obj.patentId}` };
      }
      if (obj.selectedText.length < SELECTION_MIN_CHARS) {
        return { ok: false, reason: `selectedText too short (< ${SELECTION_MIN_CHARS}): length=${obj.selectedText.length}` };
      }
      if (obj.selectedText.length > SELECTION_MAX_CHARS) {
        return { ok: false, reason: `selectedText too long (> ${SELECTION_MAX_CHARS}): length=${obj.selectedText.length}` };
      }
      return { ok: true, selection: obj };
    }

    /** Map iteration outcome → classification string (matches llm-report summary keys). */
    export function classifyIteration({ hallucinationPassed, citation, verifierStatus }) {
      if (!hallucinationPassed) return 'LLM_HALLUCINATED_SELECTION';
      if (!citation) return 'WRONG_CITATION';   // Valid selection but plugin failed to cite — falls under WRONG_CITATION in the llm-report classification space.
      if (verifierStatus === 'pass') return 'PASS';
      if (verifierStatus === 'disagree') return 'VERIFIER_DISAGREE';
      return 'WRONG_CITATION'; // verifier returned something unexpected
    }
    ```

    ```javascript
    import { spawnSync } from 'node:child_process';

    const PROJECT_ROOT = path.resolve(__dirname, '..');
    const EXTENSION_PATH = path.resolve(PROJECT_ROOT, 'dist/chrome');

    function parseArgs(argv) {
      let iterations = 5;
      for (let i = 2; i < argv.length; i++) {
        if (argv[i] === '--iterations' && argv[i+1]) {
          iterations = parseInt(argv[i+1], 10);
          if (Number.isNaN(iterations) || iterations < 1) {
            process.stderr.write(`[e2e-explore] invalid --iterations value: ${argv[i+1]}\n`);
            process.exit(2);
          }
          i++;
        } else if (argv[i] === '--help' || argv[i] === '-h') {
          process.stdout.write(
            'Usage: node scripts/e2e-explore.mjs [--iterations N]\n' +
            '  --iterations N   number of LLM iterations (default 5)\n'

        }

      return { iterations };

    function checkClaudeCli() {
      const r = spawnSync('claude', ['--version'], { encoding: 'utf8' });
      if (r.status !== 0) {
        process.stderr.write('[e2e-explore] `claude` CLI not found. Install Claude Code first.\n');

    // ---- Per-iteration ---------------------------------------------------
    async function runOneIteration({ iterationN, runId, reportPath, liveCases }) {
      const iso = new Date().toISOString();
      const tStart = Date.now();

      // 1. Pre-invocation spend cap check (LLM-06: BEFORE each invocation)
      const ledger = readLedger(LEDGER_PATH);
      const capCheck = checkSpendCap(ledger);
      if (capCheck.status === 'block') {
        process.stderr.write(`[e2e-explore] ${capCheck.message}\n`);
        return { stopAll: true };
      }
      if (capCheck.status === 'warn') {
        process.stderr.write(`[e2e-explore] ${capCheck.message}\n`);
      }

      // 2. Pre-randomize ONE candidate patent
      const candidate = liveCases[Math.floor(Math.random() * liveCases.length)];
      const patentId = candidate.id.split('-')[0]; // "US11427642-spec-short-1" → "US11427642"

      let extractResult, prompt, claudeResult, parsed, validation, hallucinationCheck;
      let extInstance = null;
      let workerHook = null;
      let citation = null, verifierVerdict = null, classification = null;
      let costUsd = 0, modelId = 'unknown';
      let rawSnippet = '';

      try {
        // 3. Extract spec text
        extractResult = await extractSpecText(patentId, { maxPages: 15 });

        // 4. Build prompt → invoke claude -p → parse
        prompt = buildPickerPrompt({
          patent: { id: patentId, category: candidate.category },
          specExcerpt: extractResult.text.slice(0, 12_000), // cap prompt body to keep token cost bounded
          bodyStartPage: extractResult.bodyStartPage,
        });
        claudeResult = await invokeClaudeP({
          systemPrompt: prompt.systemPrompt,
          userPrompt: prompt.userPrompt,
        });
        parsed = parseClaudeResponse(claudeResult);
        costUsd = parsed.costUsd ?? 0;
        modelId = parsed.modelId ?? 'unknown';
        rawSnippet = parsed.ok ? (parsed.llmText ?? '').slice(0, 2000) : (parsed.rawSnippet ?? '');

        // 5. Append to ledger ALWAYS (cost may be 0 on hard fail)

          iso, model: modelId, cost_usd: costUsd,
          tokens_in: parsed.rawJson?.usage?.input_tokens ?? 0,
          tokens_out: parsed.rawJson?.usage?.output_tokens ?? 0,
          iteration_n: iterationN, run_id: runId,
        });

        if (!parsed.ok) {
          classification = 'LLM_API_ERROR';
          appendLlmIteration(reportPath, {
            iteration_n: iterationN, iso,
            llm_selection: null, hallucination_check: null,
            citation: null, verifier_verdict: null,
            classification, cost_usd: costUsd,
            duration_ms: Date.now() - tStart,
            artifacts: [], llm_raw_response: rawSnippet,
            error_reason: parsed.errorReason,
          });
          return { stopAll: false };
        }

        // 6. Validate schema (1 retry on JSON parse error)
        validation = validateLlmSelection(parsed.llmText);
        if (!validation.ok) {
          // 1 retry per CONTEXT.md
          claudeResult = await invokeClaudeP({
            systemPrompt: prompt.systemPrompt,
            userPrompt: prompt.userPrompt + '\n\nReturn STRICT JSON ONLY. No markdown.',
          });
          parsed = parseClaudeResponse(claudeResult);
          // Even retry burns cost — record
          appendLedgerEntry(LEDGER_PATH, {
            iso: new Date().toISOString(),
            model: parsed.modelId ?? 'unknown',
            cost_usd: parsed.costUsd ?? 0,
            tokens_in: parsed.rawJson?.usage?.input_tokens ?? 0,
            tokens_out: parsed.rawJson?.usage?.output_tokens ?? 0,
            iteration_n: iterationN, run_id: runId, retry: true,
          });
          if (parsed.ok) {
            validation = validateLlmSelection(parsed.llmText);
          }
          if (!validation.ok) {
            classification = 'LLM_API_ERROR';
            appendLlmIteration(reportPath, {
              iteration_n: iterationN, iso,
              llm_selection: null, hallucination_check: null,
              citation: null, verifier_verdict: null,
              classification, cost_usd: costUsd + (parsed.costUsd ?? 0),
              duration_ms: Date.now() - tStart,
              artifacts: [], llm_raw_response: rawSnippet,
              error_reason: `schema_validation_failed: ${validation.reason}`,
            });
            return { stopAll: false };
          }
        }

        const sel = validation.selection;

        // 7. Hallucination guard (LLM-03)
        hallucinationCheck = selectionInSpec(extractResult.text, sel.selectedText);

        if (!hallucinationCheck.found) {
          classification = 'LLM_HALLUCINATED_SELECTION';
          appendLlmIteration(reportPath, {
            iteration_n: iterationN, iso,
            llm_selection: sel,
            hallucination_check: { passed: false, method: null },
            citation: null, verifier_verdict: null,
            classification, cost_usd: costUsd,
            duration_ms: Date.now() - tStart,
            artifacts: [], llm_raw_response: rawSnippet,
          });
          return { stopAll: false };
        }

        // 8. Drive the harness (LLM-01 path through to verifier)
        extInstance = await loadExtension({ extensionPath: EXTENSION_PATH });
        workerHook = await installWorkerTestModeRoute(extInstance.context, extInstance.extensionId);
        await setTriggerMode(extInstance.context, 'auto');
        await gotoPatent(extInstance.page, sel.patentId, { timeout: 30_000 });
        await selectText({ page: extInstance.page, uniqueSubstring: sel.selectedText });

        citation = obs.citation;

        // 9. Verify
        verifierVerdict = await verifyCitation({
          patentId: sel.patentId,
          selectedText: sel.selectedText,
          observedCitation: citation,
        });

        // 10. Classify
        classification = classifyIteration({
          hallucinationPassed: true,
          citation,
          verifierStatus: verifierVerdict.status,
        });

        appendLlmIteration(reportPath, {
          iteration_n: iterationN, iso,
          llm_selection: sel,
          hallucination_check: { passed: true, method: hallucinationCheck.method, needleIndex: hallucinationCheck.needleIndex },
          citation, verifier_verdict: verifierVerdict,
          classification, cost_usd: costUsd,
          duration_ms: Date.now() - tStart,
          artifacts: [], // future: screenshots on failure
          llm_raw_response: rawSnippet,
        });
        return { stopAll: false };

      } catch (err) {
        // Any unexpected error → record as LLM_API_ERROR with diagnostic
        classification = 'LLM_API_ERROR';
        appendLlmIteration(reportPath, {
          iteration_n: iterationN, iso,
          llm_selection: validation?.selection ?? null,
          hallucination_check: hallucinationCheck ? { passed: hallucinationCheck.found, method: hallucinationCheck.method ?? null } : null,
          citation, verifier_verdict: verifierVerdict,
          classification, cost_usd: costUsd,
          duration_ms: Date.now() - tStart,
          artifacts: [], llm_raw_response: rawSnippet,
          error_reason: `runtime_error: ${err.message}`,
        });
        return { stopAll: false };
      } finally {
        if (workerHook) { try { await workerHook.cleanup(); } catch {} }
        if (extInstance) { try { await extInstance.cleanup(); } catch {} }
      }
    }

    // ---- Main -----------------------------------------------------------
    async function main() {
      const { iterations } = parseArgs(process.argv);
      const claudeVer = checkClaudeCli();
      process.stdout.write(`[e2e-explore] claude ${claudeVer}\n`);
      // Initial cap check — bail early if already blocked
      const initialLedger = readLedger(LEDGER_PATH);
      const initialCap = checkSpendCap(initialLedger);
      if (initialCap.status === 'block') {
        process.stderr.write(`[e2e-explore] ${initialCap.message}\n`);
        process.exit(4);
      }
      if (initialCap.status === 'warn') {
        process.stderr.write(`[e2e-explore] ${initialCap.message}\n`);
      }
      const runId = resolveRunId();
      const reportPath = llmReportPathFor(runId);
      initLlmReport(reportPath, { run_id: runId, iterations_total: iterations });
      process.stdout.write(`[e2e-explore] run_id=${runId} iterations=${iterations} report=${reportPath}\n`);
      const liveCases = getLiveCases();

      for (let n = 1; n <= iterations; n++) {
        process.stdout.write(`[e2e-explore] iteration ${n}/${iterations}...\n`);
        const result = await runOneIteration({ iterationN: n, runId, reportPath, liveCases });
        if (result.stopAll) {
          process.stderr.write('[e2e-explore] aborting remaining iterations.\n');
          break;
        }
      }
      finalizeLlmReport(reportPath);
      process.stdout.write(`[e2e-explore] done. Report: ${reportPath}\n`);
      process.exit(0);
    }

    main().catch(err => {
      process.stderr.write(`[e2e-explore] fatal: ${err.stack || err.message}\n`);
      process.exit(5);
    });
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| LLM-supplied `selectedText` → harness `selectText({page, uniqueSubstring})` | Crosses from claude-controlled string to Playwright's `page.evaluate` (DOM API, NOT shell). |
| LLM-supplied `patentId` → `verifyCitation({patentId, ...})` → `ensureCachedPdf(patentId)` → `fetch(\`https://patents.google.com/patent/${patentId}\`)` | The patentId is interpolated into a URL string — URL encoding required to prevent injection. |
| `claude` subprocess env → spawned with explicit `ANTHROPIC_API_KEY: ''` | Prevents the developer's pay-per-token API key (if set) from being used; forces subscription pool. |
| Ledger writes during long iteration loop → file overwritten N times | Crash-safe because each write is a complete JSON document. |
