  export function topOfStackHashFromTriage(finding, rerunEntry, iteration) → 12-hex
    Inputs: JSON.stringify({
      rationale_first_30_chars: (finding.rationale ?? '').slice(0, 30),
      verifier_status: rerunEntry?.original_verdict_status ?? null,
      classification: iteration.classification ?? null,
    })
    Returns: createHash('sha256').update(input).digest('hex').substring(0, 12)

  export function findMatchingIssueDual(ghClient, fpV1, fpV2) → object | null
    1. const issuesV1 = ghClient.listOpenWithSearch(`<!-- fingerprint: ${fpV1} -->`)   ← legacy v1 marker (Phase 29 used `<!-- fingerprint: -->`)
    2. const issuesV2 = ghClient.listOpenWithSearch(`<!-- fp: ${fpV2} -->`)            ← v2 marker (D-02 new)
    Both calls execute UNCONDITIONALLY — no short-circuit (Pitfall 3).
    Returns the first issue matching EITHER marker via substring (extend findMatchingIssue to support both markers).

  export function filterFindingsForFiling(findings, rerunByIter, llmByIter) → finding[]
    For each finding:
      iter   = llmByIter.get(finding.iteration_n)
      rerun  = rerunByIter.get(finding.iteration_n)
      if (!iter) → reject
      isConfirmed   = ['critical','high'].includes(finding.severity) || rerun?.verdict === 'CONFIRMED'
      isHarnessNoise = finding.category === 'HARNESS_ERROR' || (finding.path_taken ?? '').endsWith('_parse_error')
      keep iff isConfirmed && !isHarnessNoise

  export function processTriageReport(triageReport, rerunReport, llmReport, goldenBaseline, { ghClient, runId, repo })
    1. Build Maps: rerunByIter, llmByIter
    2. filtered = filterFindingsForFiling(triageReport.findings, rerunByIter, llmByIter)
    3. For each filtered finding:
       a. caseId = iter.case_id ?? iter.llm_selection?.patentId
       b. fpV1   = fingerprint(caseId, finding.category, '')
       c. fpV2   = fingerprint(caseId, finding.category, topOfStackHashFromTriage(finding, rerunEntry, iter))
       d. existing = findMatchingIssueDual(ghClient, fpV1, fpV2)
       e. if existing → log + continue (dedup hit)
       f. else:
          goldenCitation = goldenBaseline?.[caseId]?.citation ?? null
          reproducerCmd  = `npm run e2e:explore -- --case ${caseId}`
          { title, body, labels } = buildIssuePayload({ triageFinding: finding, iteration: iter, rerunEntry, goldenCitation, reproducerCmd, fingerprint: fpV2 })
          ghClient.createIssueWithLabels(title, body, labels)
