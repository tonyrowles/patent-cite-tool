      const vulns = audit?.vulnerabilities || {};
      for (const pkg of WATCHLIST) {
        const drift = outdated[pkg];
        if (!drift) continue;
        if (NEVER_AUTO_BUMP.includes(pkg)) { skipped.push(pkg); continue; }
        const v = vulns[pkg];
        const isSecurity = v
          && ['moderate', 'high', 'critical'].includes(v.severity)
          && v.fixAvailable !== false
          && !v.fixAvailable.isSemVerMajor;
        if (isSecurity) { security.push(pkg); continue; }
        const cat = categorize(drift.current, drift.latest);
        if (cat === 'major') major.push(pkg);
        else minor.push(pkg);
      }
      return { security, minor, major, skipped };
    }

    export function writeManualBumpNote({ pkg, current, latest, notesPath = MANUAL_BUMP_NOTES_PATH }) {
      let log;
      try { log = JSON.parse(fs.readFileSync(notesPath, 'utf8')); }
      catch { log = { version: 1, notes: [] }; }
      if (log.notes.find(n => n.pkg === pkg && n.latest === latest)) return;
      log.notes.push({
        iso: new Date().toISOString(),
        pkg, current, latest,
        action: 'manual-review-required',
        reason: 'package is on NEVER_AUTO_BUMP list',
      });
      fs.mkdirSync(path.dirname(notesPath), { recursive: true });
      fs.writeFileSync(notesPath, JSON.stringify(log, null, 2) + '\n');
    }
